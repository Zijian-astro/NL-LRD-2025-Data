\f0\fs24 \cf0 -- The easiest way to find apogee is by finding objects whose spectra have\
-- been classified as apogee. This sample query searches the SpecObj\
-- table for the IDs and redshifts of objects with the class column equal to 'QSO'\
SELECT TOP 200000\
    GalSpecLine.oiii_5007_flux,\
    GalSpecLine.h_beta_flux,\
    GalSpecLine.nii_6584_flux,\
    GalSpecLine.h_alpha_flux,\
    GalSpecLine.sii_6717_flux,\
    GalSpecLine.sii_6731_flux,\
    GalSpecLine.oi_6300_flux\
FROM SpecObj\
JOIN GalSpecLine ON SpecObj.specobjid = GalSpecLine.specobjid\
WHERE SpecObj.class = 'GALAXY'\
    AND SpecObj.z BETWEEN 0.0 AND 0.5\
    AND SpecObj.snMedian_g > 18\
    AND GalSpecLine.sigma_balmer < 340\
    AND GalSpecLine.oiii_5007_flux > 5\
    AND GalSpecLine.h_beta_flux > 5\
    AND GalSpecLine.nii_6584_flux > 5\
    AND GalSpecLine.h_alpha_flux > 5\
    AND GalSpecLine.sii_6717_flux > 5\
    AND GalSpecLine.sii_6731_flux > 5\
    AND GalSpecLine.oi_6300_flux > 5}
